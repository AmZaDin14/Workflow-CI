How the image is built and pushed (since the GH runner's Docker daemon
rejects our Docker Hub PAT with "malformed HTTP Authorization header",
the build+push runs locally; the CI workflow verifies the rest):

  # Local build
  cd K3_workflow_ci/MLProject
  cp -r <mlflow_run_artifacts>/model ./model
  docker build -t amzadin14/smsml-adult:latest .
  docker push amzadin14/smsml-adult:latest

  # CI verification (the workflow queries the registry manifest endpoint
  # with a proper anonymous bearer token from auth.docker.io):
  TOKEN=$(curl -s "https://auth.docker.io/token?service=registry.docker.io&scope=repository:amzadin14/smsml-adult:pull" \
    | python -c "import json,sys; print(json.load(sys.stdin)['token'])")
  curl -sI -H "Authorization: Bearer $TOKEN" \
    "https://registry-1.docker.io/v2/amzadin14/smsml-adult/manifests/latest"
